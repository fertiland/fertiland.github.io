function preUpdateNode_xxx_1_000_to_xxx_1_100()
{
nodeType=$1
if [ “$nodeType” = “node” ] ; then
echo “preUpdate node “
else
echo “wrong node type”
fi
}

function postUpdateNode_xxx_1_100_to_xxx_1_200()
{
nodeType=$1
if [ “$nodeType” = “node” ] ; then
echo “postUpdate node “
else
echo “wrong node type”
fi
}

function dummy()
{
echo “dummy functions”
}

function preUpdate()
{
local nodeType=$1
while [[ ! “$nextVer” = “$endVer” ]]
do
case $nodeType in
“node”)
for key in ${!preUpdateNodeActions[@]}
do
if [[ “$key” = $nextVer* ]] ; then
nextVer=$(echo $key|cut -d”,” -f2)
echo $nextVer
eval ${preUpdateNodeActions[$key]} $nodeType
break
fi
done
;;
esac

   sleep 1
done

}

function postUpdate()
{
local nodeType=$1

while [[ ! "$nextVer" = "$endVer" ]]
do
   case $nodeType in
   "node")
       for key in ${!postUpdateNodeActions[@]}
       do
          if [[ "$key" = $nextVer* ]] ; then
              nextVer=$(echo $key|cut -d"," -f2)
              echo $nextVer
              eval ${postUpdateNodeActions[$key]} $nodeType
              break
          fi
       done
       ;;
   esac

   sleep 1
done

}
